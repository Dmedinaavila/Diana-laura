int distp1p2(float x1=0,float y1=0,float z1=0,float x2=0,float y2=0,float z2=0)
 {
float p1p2=sqrt((x2-x1)(x2-x1)+(y2-y1)(y2-y1)+(z2-z1)*(z2-z1));
std::cout<<p1p2;
return 0;
}
